# 命令简介 

`date` 显示系统时间，或者对日期时间进行格式化显示

# 用法

```bash
	date [-full|dt|dtms|time|timems|ss|ms|1ms|ns|1ns|fmt|d|n] 
	
	 -full       # `全部
	 -dt         # `显示日期时间到秒
	 -dtms       # `显示日期时间到毫秒
	 -dtt        # `显示日期时间到秒 结果中带T
	 -dtmst      # `显示日期时间到毫秒  结果中带T
	 -time       # `显示时间字符串到秒
	 -timems     # `显示时间字符串到毫秒
	 -ss         # `显示成秒数
	 -ms         # `显示成毫秒数
	 -1ms        # `仅仅当前秒内的毫秒数
	 -ns         # `显示成纳秒数
	 -1ns        # `仅仅当前秒内的纳秒数
	 -fmt        # `日期字符串格式化
	 -d          # `输入的日期字符串
	 -dns        # `输入纳秒数
	 -dms        # `输入毫秒数
	 -n          # `结尾不输出空行
	 -zone       # `显示当前会话时区
	 -syszone    # `显示系统时区
```
