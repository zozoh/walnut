# hMaker 编辑的页面

支持下面的元数据

```
hm_site_id : "xxx"  // 站点ID

//............................................
// 一些有用的统计信息
//............................................
// 使用的 ThingSet ID 列表
hm_list_tsid : ["uj11c22744js0p3cnlo3h1fkd8"],

// 使用的 API 列表
hm_list_api: ["/thing/get_goods"],  

// 页面可用的自定义锚点
hm_pg_anchors: []

// 页面可接受的参数
hm_pg_params: ["id"]

// 页面的控件ID列表（不包括站点组件）
hm_pg_coms:  ["rows_1", "htmlcode_1", "dynamic_1"],

//............................................
// 动态页面参数
//............................................
// 如果为 true，则表示这个页面的名称带有动态占位符
// 譬如 abc_{{id}}.html
// 那么会依次匹配这个动态占位符，并将其转换为一个 Map
// 存放在渲染上下文的 args 中
hm_pg_args: true,

//............................................
// 页面层级关系，@see hmc_crumb.md
hm_hierarchy: "xxx"

//............................................
// 提供一个保护的逻辑
//............................................
hm_pg_guard : {
    // 当前会话未登陆，重定向到何地址
    "nologin" : "login.wnml",
    // 当前会话用户没绑定手机，重定向到何地址
    "nophone" : "login.wnml",
}
```
