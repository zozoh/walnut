# Thing的元数据结构

```bash
id: ID         # 唯一ID
nm: "xxx-xx"   # 人类可识别的唯一编号
#-----------------------------
# 展示
title: "xxx"   # 标题
brief: "xxx"   # 概述
#-----------------------------
# 媒体
th_media_nb: 2       # 媒体数量
th_media_ids: [..]   # 媒体ID列表
th_media_list: [{    # 媒体列表
  id: ID,               # ID
  nm: "abc.jpg",        # 文件名
  thumb: "id:5k..5k",   # 缩略图
  mime: "image/jpeg",   # 媒体类型
  tp: "jpg",            # 扩展名
  len: 237370,          # 大小（字节）
  width: 1920,          # 宽
  height: 680,          # 高
  lm: 1591791236214,    # 最后修改
  ct: 1591791236117     # 创建时间
}]
```
